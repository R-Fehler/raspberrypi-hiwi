## WLAN

# wkit-802.1x einrichten

- wpa suplicant einstellen
  sudo nano /etc/wpa_supplicant/wpa_supplicant.conf

	 network={
			ssid="wkit-802.1x"
			key_mgmt=WPA-EAP
			pairwise=CCMP TKIP
			group=CCMP TKIP
			eap=TTLS
			phase2="auth=PAP"
			anonymous_identity="anonymous@kit.edu"
			identity="fast-fast-jma-0001"
			password="=eELx(5r"
	 }
